Admin Panel Integration Guide

How to integrate the order management system with your website

1. Setup & Installation

Follow these steps to integrate the order management system:

Step 1: Add the Order Manager Script

Include this script in your HTML pages that need order functionality:

<script src="order-manager.js"></script>

Step 2: Create Order Form

Build your order form with the necessary fields:

<form id="orderForm">
    <input type="text" id="customerName" placeholder="Full Name" required>
    <input type="email" id="email" placeholder="Email" required>
    <input type="tel" id="phone" placeholder="Phone" required>
    <textarea id="address" placeholder="Delivery Address" required></textarea>
    <textarea id="notes" placeholder="Special Notes"></textarea>
    <button type="submit">Place Order</button>
</form>

Step 3: Handle Form Submission

Add JavaScript to handle the form and save orders:

document.getElementById('orderForm').addEventListener('submit', (e) => {
    e.preventDefault();
    
    const orderData = {
        customerName: document.getElementById('customerName').value,
        email: document.getElementById('email').value,
        phone: document.getElementById('phone').value,
        address: document.getElementById('address').value,
        items: cartItems, // Your cart array
        totalAmount: cartTotal, // Your cart total
        paymentMethod: 'cash', // Or dynamic value
        notes: document.getElementById('notes').value
    };
    
    // Validate order
    const validation = orderManager.validateOrderData(orderData);
    if (!validation.isValid) {
        alert('Errors: ' + validation.errors.join(', '));
        return;
    }
    
    // Add order to system
    const newOrder = orderManager.addOrder(orderData);
    if (newOrder) {
        alert('Order placed successfully!');
        document.getElementById('orderForm').reset();
    }
});

2. Common Integration Patterns

Pattern 1: Simple Cart Checkout

// Get cart from sessionStorage
const cartItems = JSON.parse(sessionStorage.getItem('cart') || '[]');
const cartTotal = sessionStorage.getItem('cartTotal') || '0';

// Create order
const order = {
    customerName: name,
    email: email,
    phone: phone,
    address: address,
    items: cartItems,

    paymentMethod: 'cash'

// Save and process
const result = orderManager.addOrder(order);
if (result) {
    sessionStorage.removeItem('cart');
    sessionStorage.removeItem('cartTotal');
}

Pattern 3: Pre-filled Order Form

// Load customer's previous orders to pre-fill form
const customerEmail = 'customer@example.com';
const previousOrders = orderManager.getOrdersByEmail(customerEmail);

if (previousOrders.length > 0) {

    document.getElementById('email').value = lastOrder.email;
    document.getElementById('phone').value = lastOrder.phone;
    document.getElementById('address').value = lastOrder.address;
}

4. Product Page Integration

Here's how to add "Add to Cart" and "Buy Now" functionality:

Add to Cart Function

function addToCart(productName, price, quantity) {
    let cart = JSON.parse(sessionStorage.getItem('cart') || '[]');
    
    const item = {
        name: productName,
        price: parseFloat(price),
        quantity: parseInt(quantity)
    };
    
    // Check if item already in cart
    const existing = cart.find(i => i.name === productName);
    if (existing) {
        existing.quantity += parseInt(quantity);
    } else {
        cart.push(item);
    }
    
    sessionStorage.setItem('cart', JSON.stringify(cart));
    updateCartTotal();
    alert('Added to cart!');
}

function updateCartTotal() {
    const cart = JSON.parse(sessionStorage.getItem('cart') || '[]');
    const total = cart.reduce((sum, item) => 
        sum + (item.price * item.quantity), 0
    );
    sessionStorage.setItem('cartTotal', total.toString());
}

Buy Now Function

function buyNow(productName, price, quantity) {
    const cart = [{
        name: productName,
        price: parseFloat(price),
        quantity: parseInt(quantity)
    }];
    const total = price * quantity;
    
    sessionStorage.setItem('cart', JSON.stringify(cart));
    sessionStorage.setItem('cartTotal', total.toString());
    
    // Redirect to checkout
    window.location.href = 'checkout.html';
}

HTML Button Example

<button onclick="addToCart('Cloth Wash', 100, 1)">
    Add to Cart
</button>

<button onclick="buyNow('Cloth Wash', 100, 1)">
    Buy Now
</button>

5. Admin Notification System

Implement notifications when new orders are placed:

// Hook into order creation to send notifications
function createOrderWithNotification(orderData) {
    const newOrder = orderManager.addOrder(orderData);
    
    if (newOrder) {
        // Notify admin
        orderManager.notifyNewOrder(newOrder);
        
        // You can extend this to:
        // - Send email to admin
        // - Send SMS notification
        // - Create browser notification
        
        sendAdminNotification(newOrder);
        return newOrder;
    }
    return null;
}

function sendAdminNotification(order) {
    // Browser notification (if permission granted)
    if ('Notification' in window && Notification.permission === 'granted') {
        new Notification('New Order!', {
            body: `New order #${order.id} from ${order.customerName}`,
            icon: 'logo/logo.png'
        });
    }
    
    // Log to console
    console.log('New Order Notification:', order);
}
Note: To enable browser notifications, request permission from user when page loads. 
// Request notification permission
if ('Notification' in window && Notification.permission === 'default') {
    Notification.requestPermission();
}

6. Customer Order Tracking

Allow customers to track their orders:

function trackOrder(email) {
    const orders = orderManager.getOrdersByEmail(email);
    
    if (orders.length === 0) {
        return { success: false, message: 'No orders found' };
    }
    
    return {
        success: true,
        orders: orders.map(o => ({
            id: o.id,
            status: o.status,
            date: new Date(o.orderDate).toLocaleDateString(),
            total: o.totalAmount
        }))
    };
}

// Usage in HTML
function showOrderTracking() {
    const email = prompt('Enter your email:');
    if (!email) return;
    
    const result = trackOrder(email);
    
    if (result.success) {
        let html = '<h3>Your Orders</h3>';
        result.orders.forEach(order => {
            html += `<div>
                Order ${order.id}: ${order.status.toUpperCase()}
                (${order.date})
            </div>`;
        });
        document.getElementById('tracking').innerHTML = html;
    } else {
        alert(result.message);
    }
}

7. Best Practices

Data Validation

// Always validate before saving
function placeOrder(orderData) {
    const validation = orderManager.validateOrderData(orderData);
    
    if (!validation.isValid) {
        const errorMsg = validation.errors.join('\n');
        alert('Please fix errors:\n' + errorMsg);
        return false;
    }
    
    return orderManager.addOrder(orderData);
}

Error Handling

try {
    const order = orderManager.addOrder(orderData);
    if (order) {
        console.log('Order created:', order.id);
    }
} catch (error) {
    console.error('Failed to create order:', error);
    alert('An error occurred. Please try again.');
}

9. Troubleshooting

Orders Not Saving?

Check:

  1. Browser localStorage is enabled
  2. order-manager.js is loaded
  3. No JavaScript errors in console
  4. Browser storage not full

Form Not Submitting?

Check:

  1. All required fields are filled
  2. Event listener is properly attached
  3. Form ID matches in HTML and JS
  4. No conflicting JavaScript

Session Lost?

Solution:

  1. Admin session expires after 24 hours
  2. Clear browser cache if having issues
  3. Log in again to continue
